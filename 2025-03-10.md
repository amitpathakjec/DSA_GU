// Binary Search code:

public static int bS(int[] arr, int target){
    int l = 0;
    int r = arr.length - 1;
    int mid;
    
    while (l<=r){
      mid = (l+r)/2;
    
      if(arr[mid] == target){ return mid; }
      if(arr[mid] < target){ l = mid+1; }
      if(arr[mid] > target){ r = mid-1; }
    }
    return -1;
  }
